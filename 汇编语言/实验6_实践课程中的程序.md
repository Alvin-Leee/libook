## 不同的寻址方式的灵活应用

### 实验1

---

编程，将 datasg 段中每个单词的头一个字母改为大写字母

- $exp7\_1.asm$

```c
assume cs:codesg, ds:datasg

datasg segment
		db '1. file         '
		db '2. edit         '
		db '3. search       '
		db '4. view         '
		db '5. options      '
		db '6. help         '
datasg ends

codesg segment
start:	mov ax,datasg
		mov ds,ax
		
		mov bx,0
		mov cx,6		;6次循环

s:		mov al,[bx+3]	;头一个字母位置
		and al,11011111h;大写ASCII码要小，将第5位改成0
		mov [bx+3],al
		
		add bx,16

		loop s

codesg ends

end start
```



### 实验2

---

编程，将 datasg 段中每个单词改为大写字母

- $exp7\_2.asm$

```c
assume cs:codesg,ds:datasg,ss:stack;保存到栈中，就是保存在内存中

datasg segment		;4*3，dx+si
	db 'ibm             '
	db 'dec             '
	db 'dos             '
	db 'vax             '
datasg ends

stack segment
	dw 0,0,0,0,0,0,0,0
stack ends

codesg segment
start:	mov ax,datasg
	mov ds,ax
	mov ax,stack
	mov ss,ax
	mov sp,16
	
	mov bx,0
	mov cx,4	;4次大循环
	
s0:	push cx		;这里有一个分析过程
				;先是放在其余的寄存器中，但是没有一般性，因为其余寄存器会被用到
				;然后是放在内存中，如果放入很多就需要记住每个对应的内存地址，很麻烦
				;最后最优的方法是压入栈中
	mov si,0
	mov cx,3	;3次小循环

s:	mov al,[bx+si]
	and al,11011111b
	mov [bx+si],al
	
	inc si
	loop s
	
	pop cx
	add bx,16
	loop s0
	
	mov ax,4c00h
	int 21h
	
codesg ends

end start
```



### 实验3

---

编程，将 datasg 段中每个单词的前4个字母改为大写字母

- $exp7\_3.asm$

```c
assume cs:codesg,ss:stacksg,ds:datasg

stacksg segment
	dw 0,0,0,0,0,0,0,0
stacksg ends

datasg segment
	db '1. display      '
	db '2. brows        '
	db '3. replace      '
	db '4. modify       '
datasg ends

codesg segment
start:	mov ax,datasg
	mov ds,ax
	mov ax,stacksg
	mov ss,ax
	mov sp,16
	
	mov bx,0	;4*4
	mov cx,4
	
s0:	push cx
	mov si,0
	mov cx,4

s:	mov al,[bx+si+3];前3个不是字母
	and al,11011111b;注意不是add
	mov [bx+si+3],al
	
	inc si
	loop s
	
	pop cx
	add bx,16
	loop s0
	
	mov ax,4c00h
	int 21h

codesg ends

end start
```
